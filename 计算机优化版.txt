double calculateResult(int a,int b,int c)
{
	switch(a)
	{
		case 0:return (double)b+(double)c;break;
		case 1:return (double)b-(double)c;break;
		case 2:return (double)b*(double)c;break;
		case 3:return (double)b/(double)c;break;
	}
}

{
